371 -- [[ Configure LSP ]]
372 --  This function gets run when an LSP connects to a particular buffer.
373 local on_attach = function(_, bufnr)
374   -- NOTE: Remember that lua is a real programming language, and as such it is possible
375   -- to define small helper and utility functions so you don't have to repeat yourself
376   -- many times.
377   --
378   -- In this case, we create a function that lets us more easily define mappings specific
379   -- for LSP related items. It sets the mode, buffer and description for us each time.
380   local nmap = function(keys, func, desc)
381     if desc then
382       desc = 'LSP: ' .. desc
383     end
384 
385     vim.keymap.set('n', keys, func, { buffer = bufnr, desc = desc })
386   end
387 
388   nmap('<leader>rn', vim.lsp.buf.rename, '[R]e[n]ame')
389   nmap('<leader>ca', vim.lsp.buf.code_action, '[C]ode [A]ction')
390 
391   nmap('gd', vim.lsp.buf.definition, '[G]oto [D]efinition')
392   nmap('gr', require('telescope.builtin').lsp_references, '[G]oto [R]eferences')
393   nmap('gI', require('telescope.builtin').lsp_implementations, '[G]oto [I]mplementation')
394   nmap('<leader>D', vim.lsp.buf.type_definition, 'Type [D]efinition')
395   nmap('<leader>ds', require('telescope.builtin').lsp_document_symbols, '[D]ocument [S]ymbols')
396   nmap('<leader>ws', require('telescope.builtin').lsp_dynamic_workspace_symbols, '[W]orkspace [S]ymbols')
397 
398   -- See `:help K` for why this keymap
399   nmap('K', vim.lsp.buf.hover, 'Hover Documentation')
400   nmap('<C-k>', vim.lsp.buf.signature_help, 'Signature Documentation')
401 
402   -- Lesser used LSP functionality
403   nmap('gD', vim.lsp.buf.declaration, '[G]oto [D]eclaration')
404   nmap('<leader>wa', vim.lsp.buf.add_workspace_folder, '[W]orkspace [A]dd Folder')
405   nmap('<leader>wr', vim.lsp.buf.remove_workspace_folder, '[W]orkspace [R]emove Folder')
406   nmap('<leader>wl', function()
407     print(vim.inspect(vim.lsp.buf.list_workspace_folders()))
408   end, '[W]orkspace [L]ist Folders')
409 
410   -- Create a command `:Format` local to the LSP buffer
411   vim.api.nvim_buf_create_user_command(bufnr, 'Format', function(_)
412     vim.lsp.buf.format()
413   end, { desc = 'Format current buffer with LSP' })
414 end

526 -- Define a function to run the autopep8 command on the current buffer
527 function autopep8_current_file()
528   local filetype = vim.bo.filetype
529   if filetype == "python" then
530     local current_file = vim.fn.expand("%:p")
531     local command = "autopep8 " .. current_file .. " --in-place"
532     vim.fn.system(command)
533     vim.cmd(":e")
534   end
535 end
